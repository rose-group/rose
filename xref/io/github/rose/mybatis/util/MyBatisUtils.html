16  package io.github.rose.mybatis.util;
17  
18  import com.baomidou.mybatisplus.core.toolkit.StringPool;
19  import com.baomidou.mybatisplus.extension.plugins.MybatisPlusInterceptor;
20  import com.baomidou.mybatisplus.extension.plugins.inner.InnerInterceptor;
21  import io.github.rose.core.model.ToData;
22  import java.util.*;
23  import net.sf.jsqlparser.expression.Alias;
24  import net.sf.jsqlparser.schema.Column;
25  import net.sf.jsqlparser.schema.Table;
26  
27  /**
28   * @author <a href="mailto:ichensoul@gmail.com">chensoul</a>
29   * @since 0.0.1
30   */
31  public class MyBatisUtils {
32  
33      private static final String MYSQL_ESCAPE_CHARACTER = "`";
34  
35      public static <T> T getData(ToData<T> data) {
36          T object = null;
37          if (data != null) {
38              object = data.toData();
39          }
40          return object;
41      }
42  
43      public static <T> T getData(Optional<? extends ToData<T>> data) {
44          T object = null;
45          if (data.isPresent()) {
46              object = data.get().toData();
47          }
48          return object;
49      }
50  
51      public static <T> List<T> convertDataList(Collection<? extends ToData<T>> toDataList) {
52          List<T> list = Collections.emptyList();
53          if (toDataList != null && !toDataList.isEmpty()) {
54              list = new ArrayList<>();
55              for (ToData<T> object : toDataList) {
56                  if (object != null) {
57                      list.add(object.toData());
58                  }
59              }
60          }
61          return list;
62      }
63  
64      /**
65       * 将拦截器添加到链中 由于 MybatisPlusInterceptor 不支持添加拦截器，所以只能全量设置
66       *
67       * @param interceptor 链
68       * @param inner       拦截器
69       * @param index       位置
70       */
71      public static void addInterceptor(MybatisPlusInterceptor interceptor, InnerInterceptor inner, int index) {
72          List<InnerInterceptor> inners = new ArrayList<>(interceptor.getInterceptors());
73          inners.add(index, inner);
74          interceptor.setInterceptors(inners);
75      }
76  
77      /**
78       * 获得 Table 对应的表名
79       * <p>
80       * 兼容 MySQL 转义表名 `t_xxx`
81       *
82       * @param table 表
83       * @return 去除转移字符后的表名
84       */
85      public static String getTableName(Table table) {
86          String tableName = table.getName();
87          if (tableName.startsWith(MYSQL_ESCAPE_CHARACTER) && tableName.endsWith(MYSQL_ESCAPE_CHARACTER)) {
88              tableName = tableName.substring(1, tableName.length() - 1);
89          }
90          return tableName;
91      }
92  
93      /**
94       * 构建 Column 对象
95       *
96       * @param tableName  表名
97       * @param tableAlias 别名
98       * @param column     字段名
99       * @return Column 对象
100      */
101     public static Column buildColumn(String tableName, Alias tableAlias, String column) {
102         if (tableAlias != null) {
103             tableName = tableAlias.getName();
104         }
105         return new Column(tableName + StringPool.DOT + column);
106     }
107 }
