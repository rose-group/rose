16  package io.github.rose.mybatis.mq.rocketmq;
17  
18  import io.github.rose.mybatis.tenant.util.TenantContextHolder;
19  import org.apache.rocketmq.client.hook.SendMessageContext;
20  import org.apache.rocketmq.client.hook.SendMessageHook;
21  
22  import static io.github.rose.core.util.Constants.HEADER_TENANT_ID;
23  
24  /**
25   * RocketMQ 消息队列的多租户 {@link SendMessageHook} 实现类
26   * <p>
27   * Producer 发送消息时，将 {@link TenantContextHolder} 租户编号，添加到消息的 Header 中
28   *
29   * @author EnjoyIot
30   */
31  public class TenantRocketMQSendMessageHook implements SendMessageHook {
32  
33      @Override
34      public String hookName() {
35          return getClass().getSimpleName();
36      }
37  
38      @Override
39      public void sendMessageBefore(SendMessageContext sendMessageContext) {
40          String tenantId = TenantContextHolder.getTenantId();
41          if (tenantId == null) {
42              return;
43          }
44          sendMessageContext.getMessage().putUserProperty(HEADER_TENANT_ID, tenantId);
45      }
46  
47      @Override
48      public void sendMessageAfter(SendMessageContext sendMessageContext) {
49      }
50  }
