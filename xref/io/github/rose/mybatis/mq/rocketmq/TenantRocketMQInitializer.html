16  package io.github.rose.mybatis.mq.rocketmq;
17  
18  import org.apache.rocketmq.client.consumer.DefaultMQPushConsumer;
19  import org.apache.rocketmq.client.impl.consumer.DefaultMQPushConsumerImpl;
20  import org.apache.rocketmq.client.impl.producer.DefaultMQProducerImpl;
21  import org.apache.rocketmq.client.producer.DefaultMQProducer;
22  import org.apache.rocketmq.spring.core.RocketMQTemplate;
23  import org.apache.rocketmq.spring.support.DefaultRocketMQListenerContainer;
24  import org.springframework.beans.BeansException;
25  import org.springframework.beans.factory.config.BeanPostProcessor;
26  
27  /**
28   * 多租户的 RocketMQ 初始化器
29   *
30   * @author EnjoyIot
31   */
32  public class TenantRocketMQInitializer implements BeanPostProcessor {
33  
34      @Override
35      public Object postProcessAfterInitialization(Object bean, String beanName) throws BeansException {
36          if (bean instanceof DefaultRocketMQListenerContainer) {
37              DefaultRocketMQListenerContainer container = (DefaultRocketMQListenerContainer) bean;
38              initTenantConsumer(container.getConsumer());
39          } else if (bean instanceof RocketMQTemplate) {
40              RocketMQTemplate template = (RocketMQTemplate) bean;
41              initTenantProducer(template.getProducer());
42          }
43          return bean;
44      }
45  
46      private void initTenantProducer(DefaultMQProducer producer) {
47          if (producer == null) {
48              return;
49          }
50          DefaultMQProducerImpl producerImpl = producer.getDefaultMQProducerImpl();
51          if (producerImpl == null) {
52              return;
53          }
54          producerImpl.registerSendMessageHook(new TenantRocketMQSendMessageHook());
55      }
56  
57      private void initTenantConsumer(DefaultMQPushConsumer consumer) {
58          if (consumer == null) {
59              return;
60          }
61          DefaultMQPushConsumerImpl consumerImpl = consumer.getDefaultMQPushConsumerImpl();
62          if (consumerImpl == null) {
63              return;
64          }
65          consumerImpl.registerConsumeMessageHook(new TenantRocketMQConsumeMessageHook());
66      }
67  }
