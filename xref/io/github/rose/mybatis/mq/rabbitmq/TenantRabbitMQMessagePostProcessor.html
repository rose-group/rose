16  package io.github.rose.mybatis.mq.rabbitmq;
17  
18  import io.github.rose.mybatis.tenant.util.TenantContextHolder;
19  import org.apache.kafka.clients.producer.ProducerInterceptor;
20  import org.springframework.amqp.AmqpException;
21  import org.springframework.amqp.core.Message;
22  import org.springframework.amqp.core.MessagePostProcessor;
23  import org.springframework.messaging.handler.invocation.InvocableHandlerMethod;
24  
25  import static io.github.rose.core.util.Constants.HEADER_TENANT_ID;
26  
27  /**
28   * RabbitMQ 消息队列的多租户 {@link ProducerInterceptor} 实现类
29   * <p>
30   * 1. Producer 发送消息时，将 {@link TenantContextHolder} 租户编号，添加到消息的 Header 中 2. Consumer
31   * 消费消息时，将消息的 Header 的租户编号，添加到 {@link TenantContextHolder} 中，通过
32   * {@link InvocableHandlerMethod} 实现
33   */
34  public class TenantRabbitMQMessagePostProcessor implements MessagePostProcessor {
35  
36      @Override
37      public Message postProcessMessage(Message message) throws AmqpException {
38          String tenantId = TenantContextHolder.getTenantId();
39          if (tenantId != null) {
40              message.getMessageProperties().getHeaders().put(HEADER_TENANT_ID, tenantId);
41          }
42          return message;
43      }
44  }
