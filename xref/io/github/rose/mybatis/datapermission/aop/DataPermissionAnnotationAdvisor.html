16  package io.github.rose.mybatis.datapermission.aop;
17  
18  import io.github.rose.mybatis.datapermission.annotation.DataPermission;
19  import org.aopalliance.aop.Advice;
20  import org.springframework.aop.Pointcut;
21  import org.springframework.aop.support.AbstractPointcutAdvisor;
22  import org.springframework.aop.support.ComposablePointcut;
23  import org.springframework.aop.support.annotation.AnnotationMatchingPointcut;
24  
25  /**
26   * {@link DataPermission} 注解的 Advisor 实现类
27   */
28  public class DataPermissionAnnotationAdvisor extends AbstractPointcutAdvisor {
29  
30      private final Advice advice;
31  
32      private final Pointcut pointcut;
33  
34      public DataPermissionAnnotationAdvisor() {
35          this.advice = new DataPermissionAnnotationInterceptor();
36          this.pointcut = this.buildPointcut();
37      }
38  
39      @Override
40      public Advice getAdvice() {
41          return advice;
42      }
43  
44      @Override
45      public Pointcut getPointcut() {
46          return pointcut;
47      }
48  
49      protected Pointcut buildPointcut() {
50          Pointcut classPointcut = new AnnotationMatchingPointcut(DataPermission.class, true);
51          Pointcut methodPointcut = new AnnotationMatchingPointcut(null, DataPermission.class, true);
52          return new ComposablePointcut(classPointcut).union(methodPointcut);
53      }
54  }
