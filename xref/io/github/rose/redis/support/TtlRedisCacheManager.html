16  package io.github.rose.redis.support;
17  
18  import io.github.rose.core.util.StringPool;
19  import java.time.Duration;
20  import org.springframework.cache.annotation.Cacheable;
21  import org.springframework.data.redis.cache.RedisCache;
22  import org.springframework.data.redis.cache.RedisCacheConfiguration;
23  import org.springframework.data.redis.cache.RedisCacheManager;
24  import org.springframework.data.redis.cache.RedisCacheWriter;
25  import org.springframework.lang.Nullable;
26  import org.springframework.util.StringUtils;
27  
28  /**
29   * 支持自定义过期时间的 {@link RedisCacheManager} 实现类
30   * <p>
31   * 在 {@link Cacheable#cacheNames()} 格式为 "key#120s" 时，# 后面的为过期时间
32   */
33  public class TtlRedisCacheManager extends RedisCacheManager {
34      private static final int CACHE_LENGTH = 2;
35  
36      public TtlRedisCacheManager(RedisCacheWriter cacheWriter, RedisCacheConfiguration defaultCacheConfiguration) {
37          super(cacheWriter, defaultCacheConfiguration);
38      }
39  
40      public TtlRedisCacheManager(
41              RedisCacheWriter cacheWriter,
42              RedisCacheConfiguration defaultCacheConfiguration,
43              String... initialCacheNames) {
44          super(cacheWriter, defaultCacheConfiguration, true, initialCacheNames);
45      }
46  
47      @Override
48      protected RedisCache createRedisCache(String name, @Nullable RedisCacheConfiguration cacheConfig) {
49          if (!StringUtils.hasLength(name) || !name.contains(StringPool.HASH)) {
50              return super.createRedisCache(name, cacheConfig);
51          }
52          String[] names = name.split(StringPool.HASH);
53          if (names.length < CACHE_LENGTH) {
54              return super.createRedisCache(names[0], cacheConfig);
55          }
56          if (cacheConfig != null) {
57              String ttlStr = names[1].split(StringPool.COLON)[0];
58              cacheConfig = cacheConfig.entryTtl(Duration.ofSeconds(Long.parseLong(ttlStr)));
59          }
60          return super.createRedisCache(names[0], cacheConfig);
61      }
62  }
