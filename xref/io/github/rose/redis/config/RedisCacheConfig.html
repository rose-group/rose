16  package io.github.rose.redis.config;
17  
18  import io.github.rose.redis.support.TtlRedisCacheManager;
19  import org.slf4j.Logger;
20  import org.slf4j.LoggerFactory;
21  import org.springframework.boot.autoconfigure.cache.CacheManagerCustomizers;
22  import org.springframework.boot.autoconfigure.cache.CacheProperties;
23  import org.springframework.boot.context.properties.EnableConfigurationProperties;
24  import org.springframework.cache.annotation.EnableCaching;
25  import org.springframework.context.annotation.Bean;
26  import org.springframework.context.annotation.Configuration;
27  import org.springframework.data.redis.cache.CacheStatisticsCollector;
28  import org.springframework.data.redis.cache.RedisCacheConfiguration;
29  import org.springframework.data.redis.cache.RedisCacheManager;
30  import org.springframework.data.redis.cache.RedisCacheWriter;
31  import org.springframework.data.redis.connection.RedisConnectionFactory;
32  import org.springframework.data.redis.serializer.RedisSerializationContext;
33  import org.springframework.data.redis.serializer.RedisSerializer;
34  
35  @EnableCaching
36  @Configuration
37  @EnableConfigurationProperties(CacheProperties.class)
38  public class RedisCacheConfig {
39      private static final Logger log = LoggerFactory.getLogger(RedisCacheConfig.class);
40  
41      private final CacheProperties cacheProperties;
42      private final CacheManagerCustomizers cacheManagerCustomizers;
43  
44      public RedisCacheConfig(CacheProperties cacheProperties, CacheManagerCustomizers cacheManagerCustomizers) {
45          this.cacheProperties = cacheProperties;
46          this.cacheManagerCustomizers = cacheManagerCustomizers;
47      }
48  
49      @Bean
50      public RedisCacheManager redisCacheManager(RedisConnectionFactory connectionFactory) {
51          log.info("Initializing RedisCacheManager");
52  
53          RedisCacheWriter redisCacheWriter = RedisCacheWriter.nonLockingRedisCacheWriter(connectionFactory)
54                  .withStatisticsCollector(CacheStatisticsCollector.create());
55          TtlRedisCacheManager cacheManager = new TtlRedisCacheManager(
56                  redisCacheWriter,
57                  redisCacheConfiguration(),
58                  cacheProperties.getCacheNames().toArray(new String[] {}));
59          cacheManager.setTransactionAware(false);
60          return this.cacheManagerCustomizers.customize(cacheManager);
61      }
62  
63      private RedisCacheConfiguration redisCacheConfiguration() {
64          RedisCacheConfiguration config = RedisCacheConfiguration.defaultCacheConfig()
65                  .serializeValuesWith(
66                          RedisSerializationContext.SerializationPair.fromSerializer(RedisSerializer.java()));
67  
68          CacheProperties.Redis redisProperties = this.cacheProperties.getRedis();
69          if (redisProperties.getTimeToLive() != null) {
70              config = config.entryTtl(redisProperties.getTimeToLive());
71          }
72  
73          if (redisProperties.getKeyPrefix() != null) {
74              config = config.prefixCacheNameWith(redisProperties.getKeyPrefix());
75          }
76  
77          if (!redisProperties.isCacheNullValues()) {
78              config = config.disableCachingNullValues();
79          }
80  
81          if (!redisProperties.isUseKeyPrefix()) {
82              config = config.disableKeyPrefix();
83          }
84          return config;
85      }
86  }
