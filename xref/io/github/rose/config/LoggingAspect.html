16  package io.github.rose.config;
17  
18  import io.github.rose.core.json.JsonUtils;
19  import org.aspectj.lang.JoinPoint;
20  import org.aspectj.lang.ProceedingJoinPoint;
21  import org.aspectj.lang.annotation.AfterThrowing;
22  import org.aspectj.lang.annotation.Around;
23  import org.aspectj.lang.annotation.Aspect;
24  import org.aspectj.lang.annotation.Pointcut;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  import org.springframework.context.annotation.Configuration;
28  import org.springframework.context.annotation.EnableAspectJAutoProxy;
29  import org.springframework.context.annotation.Profile;
30  
31  import static io.github.rose.core.util.Constants.PROFILE_NOT_PROD;
32  
33  /**
34   * Aspect for logging execution of util and repository Spring components.
35   * <p>
36   * By default, it only runs with the "dev" profile.
37   */
38  @Profile(PROFILE_NOT_PROD)
39  @Configuration
40  @Aspect
41  @EnableAspectJAutoProxy
42  public class LoggingAspect {
43  
44      /**
45       * Pointcut that matches all repositories, services0 and Web REST endpoints.
46       */
47      @Pointcut("within(@org.springframework.stereotype.Repository *)"
48          + " || within(@org.springframework.stereotype.Service *)"
49          + " || within(@org.springframework.web.bind.annotation.RestController *)")
50      public void springBeanPointcut() {
51          // Method is empty as this is just a Pointcut, the implementations are in the
52          // advices.
53      }
54  
55      /**
56       * Pointcut that matches all Spring beans in the application's main packages.
57       */
58      @Pointcut("within(cc.chensoul..*.*Repository)" + " || within(cc.chensoul..*.*Service)"
59          + " || within(cc.chensoul..*.*Controller)")
60      public void applicationPackagePointcut() {
61          // Method is empty as this is just a Pointcut, the implementations are in the
62          // advices.
63      }
64  
65      /**
66       * Retrieves the {@link Logger} associated to the given {@link JoinPoint}.
67       *
68       * @param joinPoint join point we want the logger for.
69       * @return {@link Logger} associated to the given {@link JoinPoint}.
70       */
71      private Logger logger(JoinPoint joinPoint) {
72          return LoggerFactory.getLogger(joinPoint.getSignature().getDeclaringTypeName());
73      }
74  
75      /**
76       * Advice that logs methods throwing exceptions.
77       *
78       * @param joinPoint join point for advice.
79       * @param e         exception.
80       */
81      @AfterThrowing(pointcut = "applicationPackagePointcut() && springBeanPointcut()", throwing = "e")
82      public void logAfterThrowing(JoinPoint joinPoint, Throwable e) {
83          logger(joinPoint)
84              .error(
85                  "Exception in {}() with cause = '{}' and exception = '{}'",
86                  joinPoint.getSignature().getName(),
87                  e.getCause() != null ? String.valueOf(e.getCause()) : "NULL",
88                  e.getMessage(),
89                  e);
90      }
91  
92      /**
93       * Advice that logs when a method is entered and exited.
94       *
95       * @param joinPoint join point for advice.
96       * @return result.
97       * @throws Throwable throws {@link IllegalArgumentException}.
98       */
99      @Around("applicationPackagePointcut() && springBeanPointcut()")
100     public Object logAround(ProceedingJoinPoint joinPoint) throws Throwable {
101         Logger log = logger(joinPoint);
102         if (log.isDebugEnabled()) {
103             log.debug(
104                 "Enter {}() with arguments = {}",
105                 joinPoint.getSignature().getName(),
106                 JsonUtils.toJson(joinPoint.getArgs()));
107         }
108         try {
109             Object result = joinPoint.proceed();
110             if (log.isDebugEnabled()) {
111                 log.debug(
112                     "Exit {}() with result = {}",
113                     joinPoint.getSignature().getName(),
114                     JsonUtils.toJson(result));
115             }
116             return result;
117         } catch (IllegalArgumentException e) {
118             log.error(
119                 "Illegal argument: {} in {}()",
120                 JsonUtils.toJson(joinPoint.getArgs()),
121                 joinPoint.getSignature().getName());
122             throw e;
123         }
124     }
125 }
