16  package io.github.rose.core.json.sensitive;
17  
18  import io.github.rose.core.util.StringPool;
19  import org.apache.commons.lang3.StringUtils;
20  
21  /**
22   * @author <a href="mailto:ichensoul@gmail.com">chensoul</a>
23   * @since 0.0.1
24   */
25  public class Sensitives {
26  
27      public static final String MASK = "******";
28  
29      public static final String IPV4_MASK = ".*.*.*";
30  
31      public static final String IPV6_MASK = ":*:*:*:*:*:*:*";
32  
33      public static String deSensitive(String origin, int prefixKeep, int suffixKeep, String mask) {
34          if (StringUtils.isBlank(origin)) {
35              return StringPool.EMPTY;
36          }
37  
38          StringBuilder sb = new StringBuilder();
39          for (int i = 0, n = origin.length(); i < n; i++) {
40              if (i < prefixKeep) {
41                  sb.append(origin.charAt(i));
42                  continue;
43              }
44              if (i > (n - suffixKeep - 1)) {
45                  sb.append(origin.charAt(i));
46                  continue;
47              }
48              sb.append(mask);
49          }
50          return sb.toString();
51      }
52  
53      /**
54       * 【中文姓名】只显示最后一个汉字，其他隐藏为星号，比如：**梦
55       *
56       * @param fullName 姓名
57       * @return 结果
58       */
59      public static String chineseName(String fullName) {
60          return deSensitive(fullName, 0, 1, StringPool.STAR);
61      }
62  
63      /**
64       * 【身份证号】显示前六位, 四位，其他隐藏。共计18位或者15位，比如：340304*******1234
65       *
66       * @param id 身份证号码
67       * @return 结果
68       */
69      public static String idCardNum(String id) {
70          return deSensitive(id, 6, 4, StringPool.STAR);
71      }
72  
73      /**
74       * 【固定电话】后四位，其他隐藏，比如 ****1234
75       *
76       * @param num 固定电话
77       * @return 结果
78       */
79      public static String tel(String num) {
80          return deSensitive(num, 0, 4, StringPool.STAR);
81      }
82  
83      /**
84       * 【手机号码】前三位，后四位，其他隐藏，比如135****6810
85       *
86       * @param num 手机号码
87       * @return 结果
88       */
89      public static String phone(String num) {
90          return deSensitive(num, 3, 4, StringPool.STAR);
91      }
92  
93      /**
94       * 【地址】只显示到地区，不显示详细地址，比如：北京市海淀区****
95       *
96       * @param address 地址
97       * @return 结果
98       */
99      public static String address(String address) {
100         return deSensitive(address, 6, 0, StringPool.STAR);
101     }
102 
103     /**
104      * 【电子邮箱 邮箱前缀仅显示第一个字母，前缀其他隐藏，用星号代替，@及后面的地址显示，比如：d**@126.com
105      *
106      * @param email 电子邮箱
107      * @return 结果
108      */
109     public static String email(String email) {
110         if (email == null) {
111             return null;
112         }
113         int index = email.indexOf("@");
114         if (index <= 1) {
115             return email;
116         }
117         String preEmail = deSensitive(email.substring(0, index), 1, 0, StringPool.STAR);
118         return preEmail + email.substring(index);
119     }
120 
121     /**
122      * 【银行卡号】前六位，后四位，其他用星号隐藏每位1个星号，比如：622260**********1234
123      *
124      * @param cardNum 银行卡号
125      * @return 结果
126      */
127     public static String bankCard(String cardNum) {
128         return deSensitive(cardNum, 6, 4, StringPool.STAR);
129     }
130 
131     public static String carLicense(String carLicense) {
132         return deSensitive(carLicense, 2, 1, StringPool.STAR);
133     }
134 
135     /**
136      * 【密码】密码的全部字符都用*代替，比如：******
137      *
138      * @param secret 密码
139      * @return 结果
140      */
141     public static String secret(String secret) {
142         if (StringUtils.isBlank(secret)) {
143             return StringUtils.EMPTY;
144         }
145         return MASK;
146     }
147 
148     public static String ipv4(String ipv4) {
149         return StringUtils.substringBefore(ipv4, StringPool.DOT) + IPV4_MASK;
150     }
151 
152     public static String ipv6(String ipv6) {
153         return StringUtils.substringBefore(ipv6, StringPool.DOT) + IPV6_MASK;
154     }
155 }
