13  public abstract class Uuids {
14      // Use Java 8's built-in Base64 encoder/decoder
15      private static final Base64.Encoder URL_SAFE_ENCODER = Base64.getUrlEncoder().withoutPadding();
16      private static final Base64.Decoder URL_SAFE_DECODER = Base64.getUrlDecoder();
17  
18      private Uuids() {
19      
20      }
21  
22      public static UUID getUUID() {
23          return UuidCreator.getTimeOrderedEpoch(); // UUIDv7
24      }
25  
26      /**
27       * Generate a UUID and encode it to a URL-safe Base64 string.
28       *
29       * @return A URL-safe Base64 encoded UUID string.
30       */
31      public static String uuidToBase64(UUID uuid) {
32          ByteBuffer bb = ByteBuffer.wrap(new byte[16]);
33          bb.putLong(uuid.getMostSignificantBits());
34          bb.putLong(uuid.getLeastSignificantBits());
35          return URL_SAFE_ENCODER.encodeToString(bb.array());
36      }
37  
38      /**
39       * Decode a URL-safe Base64 string back to a UUID.
40       *
41       * @param base64 A URL-safe Base64 encoded UUID string.
42       * @return The decoded UUID.
43       */
44      public static UUID base64ToUuid(String base64) {
45          byte[] bytes = URL_SAFE_DECODER.decode(base64);
46          ByteBuffer bb = ByteBuffer.wrap(bytes);
47          long high = bb.getLong();
48          long low = bb.getLong();
49          return new UUID(high, low);
50      }
51  }
