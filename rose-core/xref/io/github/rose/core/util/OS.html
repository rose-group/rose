5   /**
6    * Operating system family.
7    * <p>
8    * Technically Solaris is UNIX, but for test purposes we are classifying it as a separate family.
9    *
10   * @since 9.2 infinispan
11   */
12  public enum OS {
13      UNIX,
14      WINDOWS,
15      SOLARIS,
16      LINUX,
17      MAC_OS;
18  
19      public static OS getCurrentOs() {
20          String os = System.getProperty("os.name").toLowerCase(Locale.CHINA);
21          if (os.contains("win")) {
22              return WINDOWS;
23          } else if (os.contains("sunos")) {
24              return SOLARIS;
25          } else if (os.contains("linux")) {
26              return LINUX;
27          } else if (os.toLowerCase(Locale.CHINA).contains("mac os")) {
28              return MAC_OS;
29          } else {
30              return UNIX;
31          }
32      }
