7   public abstract class Maps {
8       public static Map of() {
9           return Collections.unmodifiableMap(new ConcurrentHashMap());
10      }
11  
12      public static Map of(Object key, Object value) {
13          if (key == null || value == null) {
14              throw new NullPointerException("Key or value cannot be null");
15          }
16          return Collections.singletonMap(key, value);
17      }
18  
19      @SafeVarargs
20      public static Map of(Object... values) {
21          int length = values.length;
22          if (length % 2 != 0) {
23              throw new IllegalArgumentException("Keys and values must be provided in pairs");
24          }
25          Map map = new ConcurrentHashMap<>((length / 2) + 1);
26          for (int i = 0; i < length; i += 2) {
27              Object key = values[i];
28              Object value = values[i + 1];
29              if (key == null || value == null) {
30                  throw new NullPointerException("Key or value cannot be null");
31              }
32              map.put(key, value);
33          }
34          return Collections.unmodifiableMap(map);
35      }
