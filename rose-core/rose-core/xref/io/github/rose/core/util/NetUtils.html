34  public abstract class NetUtils {
35      public static final String ANY_IP4 = "0.0.0.0";
36      public static final String LOCALHOST = "localhost";
37      public static final String LOCAL_IP4 = "127.0.0.1";
38      private static final Logger log = LoggerFactory.getLogger(NetUtils.class);
39      private static final Pattern LOCAL_IP_PATTERN = Pattern.compile("127(\\.\\d{1,3}){3}$");
40      private static final Pattern IP4_PATTERN = Pattern.compile(
41          "^(\\d{1,2}|1\\d{2}|2[0-4]\\d|25[0-5])" + // 第一个数字部分
42              "(\\.(\\d{1,2}|1\\d{2}|2[0-4]\\d|25[0-5])){3}$" // 接下来的三个数字部分
43      );
44      private static final String LEGAL_LOCAL_IP_PROPERTY = "java.net.preferIPv6Addresses";
45  
46      private static InetAddress localInetAddress;
47      private static String localhostName;
48  
49      public static String getLocalAddress() {
50          return localInetAddress.getHostAddress();
51      }
52  
53      public static String getLocalhostName() {
54          if (localhostName != null) {
55              return localhostName;
56          }
57  
58          try {
59              InetAddress inetAddress = InetAddress.getLocalHost();
60              localhostName = inetAddress.getHostName();
61          } catch (UnknownHostException e) {
62              localhostName = LOCALHOST;
63          }
64          return localhostName;
65      }
66  
67      public static InetAddress getLocalInetAddress(Map<String, Integer> destHostPorts) {
68          if (localInetAddress != null) {
69              return localInetAddress;
70          }
71  
72          InetAddress inetAddress = getFirstInet4AddressByNetwork();
73  
74          if (inetAddress == null) {
75              inetAddress = getLocalAddressBySocket(destHostPorts);
76          }
77  
78          if (inetAddress == null) {
79              try {
80                  inetAddress = InetAddress.getLocalHost();
81              } catch (UnknownHostException e) {
82                  // ignore
83              }
84          }
85  
86          localInetAddress = inetAddress;
87          return inetAddress;
88      }
89  
90  
91      public static InetAddress getLocalInetAddress() {
92          return getLocalInetAddress(null);
93      }
94  
95      public static boolean isIp4Address(String ip) {
96          if (StringUtils.isBlank(ip)) {
97              return false;
98          }
99          return !ANY_IP4.equals(ip) && IP4_PATTERN.matcher(ip).matches();
100     }
101 
102     public static boolean isIp4Address(InetAddress address) {
103         if (address == null) {
104             return false;
105         }
106         String name = address.getHostAddress();
107         return (name != null && !ANY_IP4.equals(name) && !LOCAL_IP4.equals(name) && IP4_PATTERN.matcher(name).matches());
108     }
109 
110     //return ip to avoid lookup dns
111     public static String getHostName(SocketAddress socketAddress) {
112         if (socketAddress == null) {
113             return null;
114         }
115 
116         if (socketAddress instanceof InetSocketAddress) {
117             InetAddress inetAddress = ((InetSocketAddress) socketAddress).getAddress();
118             if (inetAddress != null) {
119                 return inetAddress.getHostAddress();
120             }
121         }
122 
123         return null;
124     }

126     public static String getLocalAddressByDatagram() {
127         try (final DatagramSocket socket = new DatagramSocket()) {
128             socket.connect(InetAddress.getByName("8.8.8.8"), 10002);
129             return socket.getLocalAddress().getHostAddress();
130         } catch (Exception e) {
131             log.error("Failed to retrieving ip address.", e);
132         }
133         return null;
134     }

136     public static int getAvailablePort() {
137         for (int i = 0; i < 50; i++) {
138             try (ServerSocket serverSocket = new ServerSocket(0)) {
139                 int port = serverSocket.getLocalPort();
140                 if (port != 0) {
141                     return port;
142                 }
143             } catch (IOException ignored) {
144             }
145         }
146 
147         throw new RuntimeException("Could not find a free permitted port on the machine.");
148     }

150     public static String resolveAddress(String fqdn) {
151         String ip = null;
152         try {
153             InetAddress address = InetAddress.getByName(fqdn);
154             ip = address.getHostAddress();
155         } catch (UnknownHostException e) {
156             log.error("UnknownHostException " + fqdn, e);
157         }
158         return ip;
159     }
160 
161     public static URI resolveAddress(URI uri) {
162         // convert the uri to URL.
163         try {
164             URL url = uri.toURL();
165             String host = url.getHost();
166             String ip = resolveAddress(host);
167             if (ip != null) {
168                 url = new URL(url.getProtocol(), ip, url.getPort(), url.getFile());
169                 uri = url.toURI();
170             }
171         } catch (MalformedURLException | URISyntaxException e) {
172             throw new RuntimeException(e);
173         }
174         return uri;
175     }

177     private static InetAddress getLocalAddressBySocket(Map<String, Integer> destHostPorts) {
178         if (destHostPorts == null || destHostPorts.size() == 0) {
179             return null;
180         }
181 
182         for (Map.Entry<String, Integer> entry : destHostPorts.entrySet()) {
183             String host = entry.getKey();
184             int port = entry.getValue();
185             try {
186                 Socket socket = new Socket();
187                 try {
188                     SocketAddress socketAddress = new InetSocketAddress(host, port);
189                     socket.connect(socketAddress, 1000);
190                     return socket.getLocalAddress();
191                 } finally {
192                     try {
193                         socket.close();
194                     } catch (Throwable e) {
195                     }
196                 }
197             } catch (Exception e) {
198                 log.error("Failed to retrieve local address by connecting to dest host:port({}:{}) false, e={}", host, port, e);
199             }
200         }
201         return null;
202     }

204     /**
205      * 参考 nacos-client NetUtils
206      *
207      * @return
208      */
209     private static InetAddress getFirstInet4AddressByNetwork() {
210         InetAddress result = null;
211 
212         try {
213             Enumeration<NetworkInterface> interfaces = NetworkInterface.getNetworkInterfaces();
214             int lowest = Integer.MAX_VALUE;
215 
216             while (interfaces.hasMoreElements()) {
217                 NetworkInterface network = interfaces.nextElement();
218                 if (network.isUp() && (network.getIndex() < lowest || result == null)) {
219                     lowest = network.getIndex();
220                     Enumeration<InetAddress> addresses = network.getInetAddresses();
221                     while (addresses.hasMoreElements()) {
222                         InetAddress address = addresses.nextElement();
223                         boolean isLegalIpVersion = Boolean.parseBoolean(System.getProperty(LEGAL_LOCAL_IP_PROPERTY))
224                             ? address instanceof Inet6Address : address instanceof Inet4Address;
225                         if (isLegalIpVersion && !address.isLoopbackAddress()) {
226                             result = address;
227                         }
228                     }
229                 }
230             }
231         } catch (Exception e) {
232             //ignore
233         }
234         return result;
235     }
