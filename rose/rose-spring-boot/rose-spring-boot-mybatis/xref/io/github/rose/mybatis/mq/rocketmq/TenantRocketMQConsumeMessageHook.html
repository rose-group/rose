16  package io.github.rose.mybatis.mq.rocketmq;
17  
18  import com.baomidou.mybatisplus.core.toolkit.Assert;
19  import com.baomidou.mybatisplus.core.toolkit.StringUtils;
20  import io.github.rose.mybatis.tenant.util.TenantContextHolder;
21  import org.apache.rocketmq.client.hook.ConsumeMessageContext;
22  import org.apache.rocketmq.client.hook.ConsumeMessageHook;
23  import org.apache.rocketmq.common.message.MessageExt;
24  import org.springframework.messaging.handler.invocation.InvocableHandlerMethod;
25  
26  import java.util.List;
27  
28  import static io.github.rose.core.util.Constants.HEADER_TENANT_ID;
29  
30  /**
31   * RocketMQ 消息队列的多租户 {@link ConsumeMessageHook} 实现类
32   * <p>
33   * Consumer 消费消息时，将消息的 Header 的租户编号，添加到 {@link TenantContextHolder} 中，通过
34   * {@link InvocableHandlerMethod} 实现
35   */
36  public class TenantRocketMQConsumeMessageHook implements ConsumeMessageHook {
37  
38      @Override
39      public String hookName() {
40          return getClass().getSimpleName();
41      }
42  
43      @Override
44      public void consumeMessageBefore(ConsumeMessageContext context) {
45          // 校验，消息必须是单条，不然设置租户可能不正确
46          List<MessageExt> messages = context.getMsgList();
47          Assert.isTrue(messages.size() == 1, "消息条数({})不正确", messages.size());
48          // 设置租户编号
49          String tenantId = messages.get(0).getUserProperty(HEADER_TENANT_ID);
50          if (StringUtils.isNotEmpty(tenantId)) {
51              TenantContextHolder.setTenantId(tenantId);
52          }
53      }
54  
55      @Override
56      public void consumeMessageAfter(ConsumeMessageContext context) {
57          TenantContextHolder.clear();
58      }
59  }
