16  package io.github.rose.redis.mq.job;
17  
18  import io.github.rose.redis.mq.RedisMQTemplate;
19  import io.github.rose.redis.mq.stream.AbstractRedisStreamMessageListener;
20  import java.util.List;
21  import java.util.Map;
22  import java.util.Objects;
23  import org.redisson.api.RLock;
24  import org.redisson.api.RedissonClient;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  import org.springframework.data.domain.Range;
28  import org.springframework.data.redis.connection.stream.*;
29  import org.springframework.data.redis.core.StreamOperations;
30  import org.springframework.scheduling.annotation.Scheduled;
31  import org.springframework.util.CollectionUtils;
32  
33  /**
34   * 这个任务用于处理，crash 之后的消费者未消费完的消息
35   */
36  public class RedisPendingMessageResendJob {
37      private static final Logger log = LoggerFactory.getLogger(RedisPendingMessageResendJob.class);
38  
39      private static final String LOCK_KEY = "mq:pending:msg:lock";
40  
41      /**
42       * 消息超时时间，默认 5 分钟
43       * <p>
44       * 1. 超时的消息才会被重新投递 2. 由于定时任务 1 分钟一次，消息超时后不会被立即重投，极端情况下消息5分钟过期后，再等 1 分钟才会被扫瞄到
45       */
46      private static final int EXPIRE_TIME = 5 * 60;
47  
48      private final List<AbstractRedisStreamMessageListener<?>> listeners;
49  
50      private final RedisMQTemplate redisTemplate;
51  
52      private final String groupName;
53  
54      private final RedissonClient redissonClient;
55  
56      public RedisPendingMessageResendJob(
57              List<AbstractRedisStreamMessageListener<?>> listeners,
58              RedisMQTemplate redisTemplate,
59              String groupName,
60              RedissonClient redissonClient) {
61          this.listeners = listeners;
62          this.redisTemplate = redisTemplate;
63          this.groupName = groupName;
64          this.redissonClient = redissonClient;
65      }
66  
67      /**
68       * 一分钟执行一次,这里选择每分钟的35秒执行，是为了避免整点任务过多的问题
69       */
70      @Scheduled(cron = "35 * * * * ?")
71      public void messageResend() {
72          RLock lock = redissonClient.getLock(LOCK_KEY);
73          // 尝试加锁
74          if (lock.tryLock()) {
75              try {
76                  execute();
77              } catch (Exception ex) {
78                  log.error("[messageResend][执行异常]", ex);
79              } finally {
80                  lock.unlock();
81              }
82          }
83      }
84  
85      /**
86       * 执行清理逻辑
87       *
88       * @see <a href="https://gitee.com/zhijiantianya/ruoyi-vue-pro/pulls/480/files">讨论</a>
89       */
90      private void execute() {
91          StreamOperations<String, Object, Object> ops =
92                  redisTemplate.getRedisTemplate().opsForStream();
93          listeners.forEach(listener -> {
94              PendingMessagesSummary pendingMessagesSummary =
95                      Objects.requireNonNull(ops.pending(listener.getStreamKey(), groupName));
96              // 每个消费者的 pending 队列消息数量
97              Map<String, Long> pendingMessagesPerConsumer = pendingMessagesSummary.getPendingMessagesPerConsumer();
98              pendingMessagesPerConsumer.forEach((consumerName, pendingMessageCount) -> {
99                  log.info("[processPendingMessage][消费者({}) 消息数量({})]", consumerName, pendingMessageCount);
100                 // 每个消费者的 pending消息的详情信息
101                 PendingMessages pendingMessages = ops.pending(
102                         listener.getStreamKey(),
103                         Consumer.from(groupName, consumerName),
104                         Range.unbounded(),
105                         pendingMessageCount);
106                 if (pendingMessages.isEmpty()) {
107                     return;
108                 }
109                 pendingMessages.forEach(pendingMessage -> {
110                     // 获取消息上一次传递到 consumer 的时间,
111                     long lastDelivery =
112                             pendingMessage.getElapsedTimeSinceLastDelivery().getSeconds();
113                     if (lastDelivery < EXPIRE_TIME) {
114                         return;
115                     }
116                     // 获取指定 id 的消息体
117                     List<MapRecord<String, Object, Object>> records = ops.range(
118                             listener.getStreamKey(),
119                             Range.of(
120                                     Range.Bound.inclusive(pendingMessage.getIdAsString()),
121                                     Range.Bound.inclusive(pendingMessage.getIdAsString())));
122                     if (CollectionUtils.isEmpty(records)) {
123                         return;
124                     }
125                     // 重新投递消息
126                     redisTemplate
127                             .getRedisTemplate()
128                             .opsForStream()
129                             .add(StreamRecords.newRecord()
130                                     .ofObject(records.get(0).getValue()) // 设置内容
131                                     .withStreamKey(listener.getStreamKey()));
132                     // ack 消息消费完成
133                     redisTemplate.getRedisTemplate().opsForStream().acknowledge(groupName, records.get(0));
134                     log.info(
135                             "[processPendingMessage][消息({})重新投递成功]",
136                             records.get(0).getId());
137                 });
138             });
139         });
140     }
