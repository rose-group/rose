16  package io.github.rose.redis.config;
17  
18  import io.github.rose.redis.service.RedisService;
19  import org.redisson.spring.starter.RedissonAutoConfigurationV2;
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  import org.springframework.boot.autoconfigure.AutoConfigureBefore;
23  import org.springframework.boot.autoconfigure.cache.CacheProperties;
24  import org.springframework.boot.autoconfigure.condition.ConditionalOnBean;
25  import org.springframework.boot.autoconfigure.condition.ConditionalOnClass;
26  import org.springframework.boot.autoconfigure.data.redis.RedisAutoConfiguration;
27  import org.springframework.boot.context.properties.EnableConfigurationProperties;
28  import org.springframework.context.annotation.Bean;
29  import org.springframework.context.annotation.Configuration;
30  import org.springframework.data.redis.connection.RedisConfiguration;
31  import org.springframework.data.redis.connection.RedisConnectionFactory;
32  import org.springframework.data.redis.core.RedisOperations;
33  import org.springframework.data.redis.core.RedisTemplate;
34  import org.springframework.data.redis.core.script.DefaultRedisScript;
35  import org.springframework.data.redis.serializer.RedisSerializer;
36  
37  @Configuration
38  @AutoConfigureBefore({RedisConfiguration.class, RedisAutoConfiguration.class, RedissonAutoConfigurationV2.class})
39  @EnableConfigurationProperties({CacheProperties.class})
40  public class RedisTemplateConfig {
41      private static final Logger log = LoggerFactory.getLogger(RedisTemplateConfig.class);
42  
43      public RedisTemplateConfig() {
44          log.info("Initializing RedisTemplateConfig");
45      }
46  
47      @Bean
48      @ConditionalOnClass(RedisOperations.class)
49      public RedisTemplate<String, Object> redisTemplate(RedisConnectionFactory redisConnectionFactory) {
50          RedisTemplate<String, Object> redisTemplate = new RedisTemplate<>();
51          redisTemplate.setConnectionFactory(redisConnectionFactory);
52          redisTemplate.setKeySerializer(RedisSerializer.string());
53          redisTemplate.setValueSerializer(RedisSerializer.java());
54          redisTemplate.setHashKeySerializer(RedisSerializer.string());
55          redisTemplate.setHashValueSerializer(RedisSerializer.java());
56          redisTemplate.afterPropertiesSet();
57          return redisTemplate;
58      }
59  
60      @Bean
61      @ConditionalOnBean(name = "redisTemplate")
62      public RedisService redisService() {
63          return new RedisService();
64      }
65  
66      @Bean
67      public DefaultRedisScript<Long> limitScript() {
68          DefaultRedisScript<Long> redisScript = new DefaultRedisScript<>();
69          redisScript.setScriptText(limitScriptText());
70          redisScript.setResultType(Long.class);
71          return redisScript;
72      }
73  
74      private String limitScriptText() {
75          return "local key = KEYS[1]\n" + "local count = tonumber(ARGV[1])\n" + "local time = tonumber(ARGV[2])\n"
76                  + "local current = redis.call('get', key);\n" + "if current and tonumber(current) > count then\n"
77                  + "    return tonumber(current);\n" + "end\n" + "current = redis.call('incr', key)\n"
78                  + "if tonumber(current) == 1 then\n" + "    redis.call('expire', key, time)\n" + "end\n"
79                  + "return tonumber(current);";
80      }
81  }
