16  package io.github.rose.config;
17  
18  import io.github.rose.core.spring.SpringContextHolder;
19  import io.github.rose.core.spring.factory.YamlPropertySourceFactory;
20  import io.github.rose.core.util.StringPool;
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  import org.springframework.boot.autoconfigure.AutoConfiguration;
24  import org.springframework.boot.web.context.WebServerInitializedEvent;
25  import org.springframework.context.annotation.PropertySource;
26  import org.springframework.context.event.EventListener;
27  import org.springframework.core.Ordered;
28  import org.springframework.core.annotation.Order;
29  
30  @AutoConfiguration
31  @Order(value = Ordered.HIGHEST_PRECEDENCE)
32  @PropertySource(value = "classpath:common-config.yml", factory = YamlPropertySourceFactory.class)
33  public class EnvironmentConfiguration {
34      private static final Logger log = LoggerFactory.getLogger(EnvironmentConfiguration.class);
35  
36      @Order
37      @EventListener(WebServerInitializedEvent.class)
38      public void afterStart(WebServerInitializedEvent event) {
39          String appName = SpringContextHolder.getApplicationName();
40          int localPort = event.getWebServer().getPort();
41          String profiles = String.join(StringPool.COMMA, SpringContextHolder.getActiveProfiles());
42          log.info("Application {} finish to start with port {} and {} profile", appName, localPort, profiles);
43      }
44  }
