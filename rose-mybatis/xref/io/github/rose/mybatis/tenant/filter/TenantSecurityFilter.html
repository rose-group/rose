16  package io.github.rose.mybatis.tenant.filter;
17  
18  import io.github.rose.core.spring.WebUtils;
19  import io.github.rose.core.util.RestResponse;
20  import io.github.rose.mybatis.tenant.util.TenantContextHolder;
21  import io.github.rose.security.util.SecurityUser;
22  import io.github.rose.security.util.SecurityUtils;
23  import java.io.IOException;
24  import java.util.Set;
25  import javax.servlet.FilterChain;
26  import javax.servlet.ServletException;
27  import javax.servlet.http.HttpServletRequest;
28  import javax.servlet.http.HttpServletResponse;
29  import org.slf4j.Logger;
30  import org.slf4j.LoggerFactory;
31  import org.springframework.http.HttpStatus;
32  import org.springframework.util.AntPathMatcher;
33  import org.springframework.web.filter.OncePerRequestFilter;
34  
35  /**
36   * 多租户 Security Web 过滤器 1. 如果是登陆的用户，校验是否有权限访问该租户，避免越权问题。 2. 如果请求未带租户的编号，检查是否是忽略的
37   * URL，否则也不允许访问。 3. 校验租户是合法，例如说被禁用、到期
38   *
39   * @author EnjoyIot
40   */
41  public class TenantSecurityFilter extends OncePerRequestFilter {
42      private static final Logger log = LoggerFactory.getLogger(TenantSecurityFilter.class);
43  
44      private static final AntPathMatcher pathMatcher = new AntPathMatcher();
45  
46      private final Set<String> ignoreUrls;
47  
48      public TenantSecurityFilter(Set<String> ignoreUrls) {
49          this.ignoreUrls = ignoreUrls;
50      }
51  
52      @Override
53      protected void doFilterInternal(HttpServletRequest request, HttpServletResponse response, FilterChain chain)
54              throws ServletException, IOException {
55          String tenantId = TenantContextHolder.getTenantId();
56  
57          // 如果非允许忽略租户的 URL，则校验租户是否合法
58          if (!isIgnoreUrl(request)) {
59              // 如果请求未带租户的编号，不允许访问。
60              if (tenantId == null) {
61                  log.error("[doFilterInternal][URL({}/{}) 未传递租户编号]", request.getRequestURI(), request.getMethod());
62                  WebUtils.renderJson(HttpStatus.FORBIDDEN.value(), RestResponse.error("请求的租户标识未传递"));
63                  return;
64              }
65              // // 校验租户是合法，例如说被禁用、到期
66              // try {
67              // tenantFrameworkService.validTenant(tenantId);
68              // } catch (Throwable ex) {
69              // CommonResult<?> result =
70              // globalExceptionHandler.allExceptionHandler(request, ex);
71              // ServletUtils.writeJSON(response, result);
72              // return;
73              // }
74          } else {
75              // 如果是允许忽略租户的 URL，若未传递租户编号，则默认忽略租户编号，避免报错
76              if (tenantId == null) {
77                  TenantContextHolder.setIgnore(true);
78              }
79          }
80  
81          // 1. 登陆的用户，校验是否有权限访问该租户，避免越权问题。
82          SecurityUser user = SecurityUtils.getCurrentUser();
83          if (user != null) {
84              // 如果获取不到租户编号，则尝试使用登陆用户的租户编号
85              if (tenantId == null) {
86                  tenantId = user.getTenants().get(0);
87                  TenantContextHolder.setTenantId(tenantId);
88                  // 如果传递了租户编号，则进行比对租户编号，避免越权问题
89              } else if (!user.getTenants().contains(TenantContextHolder.getTenantId())) {
90                  log.error(
91                          "用户{}越权访问租户({}) URL({}/{})]",
92                          user.getUsername(),
93                          TenantContextHolder.getTenantId(),
94                          request.getRequestURI(),
95                          request.getMethod());
96                  WebUtils.renderJson(HttpStatus.FORBIDDEN.value(), RestResponse.error("您无权访问该租户的数据"));
97                  return;
98              }
99          }
100         chain.doFilter(request, response);
101     }
102 
103     private boolean isIgnoreUrl(HttpServletRequest request) {
104         // 快速匹配，保证性能
105         if (!ignoreUrls.contains(request.getRequestURI())) {
106             return true;
107         }
108         // 逐个 Ant 路径匹配
109         for (String url : ignoreUrls) {
110             if (pathMatcher.match(url, request.getRequestURI())) {
111                 return true;
112             }
113         }
114         return false;
115     }
116 }
