16  package io.github.rose.mybatis.tenant.handler;
17  
18  import com.baomidou.mybatisplus.extension.plugins.handler.TenantLineHandler;
19  import com.baomidou.mybatisplus.extension.toolkit.SqlParserUtils;
20  import io.github.rose.mybatis.tenant.util.TenantContextHolder;
21  import net.sf.jsqlparser.expression.Expression;
22  import net.sf.jsqlparser.expression.StringValue;
23  
24  import java.util.HashSet;
25  import java.util.Locale;
26  import java.util.Set;
27  
28  /**
29   * @author <a href="mailto:ichensoul@gmail.com">chensoul</a>
30   * @since 0.0.1
31   */
32  public class DefaultTenantLineHandler implements TenantLineHandler {
33  
34      private final Set<String> ignoredTables = new HashSet<>();
35  
36      public DefaultTenantLineHandler(Set<String> ignoredTables) {
37          // 不同 DB 下，大小写的习惯不同，所以需要都添加进去
38          ignoredTables.forEach(table -> {
39              this.ignoredTables.add(table.toLowerCase(Locale.getDefault()));
40              this.ignoredTables.add(table.toUpperCase(Locale.getDefault()));
41          });
42      }
43  
44      @Override
45      public Expression getTenantId() {
46          return new StringValue(TenantContextHolder.getTenantId());
47      }
48  
49      @Override
50      public boolean ignoreTable(String tableName) {
51          return TenantContextHolder.isIgnored() // 情况一，全局忽略多租户
52              || ignoredTables.contains(SqlParserUtils.removeWrapperSymbol(tableName)); // 情况二，忽略多租户的表
53      }
54  }
