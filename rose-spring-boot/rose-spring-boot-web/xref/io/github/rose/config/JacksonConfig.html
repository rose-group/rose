3   import com.fasterxml.jackson.databind.DeserializationFeature;
4   import com.fasterxml.jackson.databind.SerializationFeature;
5   import org.springframework.boot.autoconfigure.jackson.Jackson2ObjectMapperBuilderCustomizer;
6   import org.springframework.context.annotation.Bean;
7   import org.springframework.context.annotation.Configuration;
8   
9   @Configuration
10  public class JacksonConfig {
11      @Bean
12      public Jackson2ObjectMapperBuilderCustomizer jacksonCustomizer() {
13          return builder -> builder
14              .featuresToDisable(
15                  DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES,
16                  DeserializationFeature.ACCEPT_FLOAT_AS_INT,
17                  SerializationFeature.WRITE_DATES_AS_TIMESTAMPS
18              ).findModulesViaServiceLoader(true);
19      }
