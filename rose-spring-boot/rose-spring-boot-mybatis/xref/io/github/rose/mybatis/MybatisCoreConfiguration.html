16  package io.github.rose.mybatis;
17  
18  import com.baomidou.mybatisplus.autoconfigure.MybatisPlusAutoConfiguration;
19  import com.baomidou.mybatisplus.core.handlers.MetaObjectHandler;
20  import com.baomidou.mybatisplus.core.incrementer.DefaultIdentifierGenerator;
21  import com.baomidou.mybatisplus.core.incrementer.IdentifierGenerator;
22  import com.baomidou.mybatisplus.core.injector.DefaultSqlInjector;
23  import com.baomidou.mybatisplus.core.injector.ISqlInjector;
24  import com.baomidou.mybatisplus.extension.parser.JsqlParserGlobal;
25  import com.baomidou.mybatisplus.extension.parser.cache.JdkSerialCaffeineJsqlParseCache;
26  import com.baomidou.mybatisplus.extension.plugins.MybatisPlusInterceptor;
27  import com.baomidou.mybatisplus.extension.plugins.inner.PaginationInnerInterceptor;
28  import io.github.rose.core.util.NetUtils;
29  import io.github.rose.mybatis.extension.interceptor.DefaultMetaObjectHandler;
30  import io.github.rose.mybatis.mq.rabbitmq.TenantRabbitMQInitializer;
31  import io.github.rose.mybatis.mq.redis.TenantRedisMessageInterceptor;
32  import io.github.rose.mybatis.mq.rocketmq.TenantRocketMQInitializer;
33  import org.springframework.boot.autoconfigure.AutoConfiguration;
34  import org.springframework.boot.autoconfigure.condition.ConditionalOnClass;
35  import org.springframework.boot.autoconfigure.condition.ConditionalOnMissingBean;
36  import org.springframework.context.annotation.Bean;
37  
38  import java.util.concurrent.TimeUnit;
39  
40  /**
41   * @author <a href="mailto:ichensoul@gmail.com">chensoul</a>
42   * @since 0.0.1
43   */
44  // 先于 MyBatis Plus 自动配置，避免 @MapperScan 可能扫描不到 Mapper 打印 warn 日志
45  @AutoConfiguration(before = MybatisPlusAutoConfiguration.class)
46  public class MybatisCoreConfiguration {
47  
48      static {
49          // 动态 SQL 智能优化支持本地缓存加速解析，更完善的租户复杂 XML 动态 SQL 支持，静态注入缓存
50          JsqlParserGlobal.setJsqlParseCache(new JdkSerialCaffeineJsqlParseCache(
51              (cache) -> cache.maximumSize(1024).expireAfterWrite(5, TimeUnit.SECONDS)));
52      }
53  
54      @Bean
55      public IdentifierGenerator idGenerator() {
56          // 通过本地 生成 workerId 和 dataCenterId
57          return new DefaultIdentifierGenerator(NetUtils.getLocalInetAddress());
58      }
59  
60      @Bean
61      public MybatisPlusInterceptor mybatisPlusInterceptor() {
62          MybatisPlusInterceptor mybatisPlusInterceptor = new MybatisPlusInterceptor();
63          PaginationInnerInterceptor paginationInnerInterceptor = new PaginationInnerInterceptor();
64          paginationInnerInterceptor.setMaxLimit(1000L);
65          mybatisPlusInterceptor.addInnerInterceptor(paginationInnerInterceptor);
66          return mybatisPlusInterceptor;
67      }
68  
69      @Bean
70      @ConditionalOnMissingBean
71      public ISqlInjector sqlInjector() {
72          return new DefaultSqlInjector();
73      }
74  
75      @Bean
76      @ConditionalOnMissingBean
77      public MetaObjectHandler metaObjectHandler() {
78          return new DefaultMetaObjectHandler();
79      }
80  
81      @Bean
82      public TenantRedisMessageInterceptor tenantRedisMessageInterceptor() {
83          return new TenantRedisMessageInterceptor();
84      }
85  
86      @Bean
87      @ConditionalOnClass(name = "org.springframework.amqp.rabbit.core.RabbitTemplate")
88      public TenantRabbitMQInitializer tenantRabbitMQInitializer() {
89          return new TenantRabbitMQInitializer();
90      }
91  
92      @Bean
93      @ConditionalOnClass(name = "org.apache.rocketmq.spring.core.RocketMQTemplate")
94      public TenantRocketMQInitializer tenantRocketMQInitializer() {
95          return new TenantRocketMQInitializer();
96      }
97  }
