16  package io.github.rose.mybatis.mq.kafka;
17  
18  import org.slf4j.Logger;
19  import org.slf4j.LoggerFactory;
20  import org.springframework.boot.SpringApplication;
21  import org.springframework.boot.env.EnvironmentPostProcessor;
22  import org.springframework.core.env.ConfigurableEnvironment;
23  import org.springframework.util.StringUtils;
24  
25  /**
26   * 多租户的 Kafka 的 {@link EnvironmentPostProcessor} 实现类
27   * <p>
28   * Kafka Producer 发送消息时，增加 {@link TenantKafkaProducerInterceptor} 拦截器
29   */
30  public class TenantKafkaEnvironmentPostProcessor implements EnvironmentPostProcessor {
31      private static final Logger log = LoggerFactory.getLogger(TenantKafkaEnvironmentPostProcessor.class);
32  
33      private static final String PROPERTY_KEY_INTERCEPTOR_CLASSES =
34              "spring.kafka.producer.properties.interceptor.classes";
35  
36      @Override
37      public void postProcessEnvironment(ConfigurableEnvironment environment, SpringApplication application) {
38          // 添加 TenantKafkaProducerInterceptor 拦截器
39          try {
40              String value = environment.getProperty(PROPERTY_KEY_INTERCEPTOR_CLASSES);
41              if (StringUtils.isEmpty(value)) {
42                  value = TenantKafkaProducerInterceptor.class.getName();
43              } else {
44                  value += "," + TenantKafkaProducerInterceptor.class.getName();
45              }
46              environment.getSystemProperties().put(PROPERTY_KEY_INTERCEPTOR_CLASSES, value);
47          } catch (NoClassDefFoundError ignore) {
48              // 如果触发 NoClassDefFoundError 异常，说明 TenantKafkaProducerInterceptor 类不存在，即没引入
49              // kafka-spring 依赖
50          }
51      }
52  }
