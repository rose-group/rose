37  import java.lang.reflect.InvocationHandler;

48  /**
49   * 支持自动降级注入 重写 {@link com.alibaba.cloud.sentinel.feign.SentinelInvocationHandler}
50   */
51  public class SentinelInvocationHandler implements InvocationHandler {
52      public static final String EQUALS = "equals";
53      public static final String HASH_CODE = "hashCode";
54      public static final String TO_STRING = "toString";
55      private static final Logger log = LoggerFactory.getLogger(SentinelInvocationHandler.class);
56      private final Target<?> target;
57  
58      private final Map<Method, InvocationHandlerFactory.MethodHandler> dispatch;
59  
60      private FallbackFactory<?> fallbackFactory;
61  
62      private Map<Method, Method> fallbackMethodMap;
63  
64      SentinelInvocationHandler(
65          Target<?> target,
66          Map<Method, InvocationHandlerFactory.MethodHandler> dispatch,
67          FallbackFactory<?> fallbackFactory) {
68          this.target = checkNotNull(target, "target");
69          this.dispatch = checkNotNull(dispatch, "dispatch");
70          this.fallbackFactory = fallbackFactory;
71          this.fallbackMethodMap = toFallbackMethod(dispatch);
72      }
73  
74      SentinelInvocationHandler(Target<?> target, Map<Method, InvocationHandlerFactory.MethodHandler> dispatch) {
75          this.target = checkNotNull(target, "target");
76          this.dispatch = checkNotNull(dispatch, "dispatch");
77      }
78  
79      static Map<Method, Method> toFallbackMethod(Map<Method, InvocationHandlerFactory.MethodHandler> dispatch) {
80          Map<Method, Method> result = new LinkedHashMap<>();
81          for (Method method : dispatch.keySet()) {
82              ReflectionUtils.makeAccessible(method);
83              result.put(method, method);
84          }
85          return result;
86      }
87  
88      @Override
89      public Object invoke(final Object proxy, final Method method, final Object[] args) throws Throwable {
90          if (EQUALS.equals(method.getName())) {
91              try {
92                  Object otherHandler = args.length > 0 && args[0] != null ? Proxy.getInvocationHandler(args[0]) : null;
93                  return equals(otherHandler);
94              } catch (IllegalArgumentException e) {
95                  return false;
96              }
97          } else if (HASH_CODE.equals(method.getName())) {
98              return hashCode();
99          } else if (TO_STRING.equals(method.getName())) {
100             return toString();
101         }
102 
103         Object result;
104         InvocationHandlerFactory.MethodHandler methodHandler = this.dispatch.get(method);
105         // only handle by HardCodedTarget
106         if (target instanceof Target.HardCodedTarget) {
107             Target.HardCodedTarget<?> hardCodedTarget = (Target.HardCodedTarget) target;
108             MethodMetadata methodMetadata = SentinelContractHolder.METADATA_MAP.get(
109                 hardCodedTarget.type().getName() + Feign.configKey(hardCodedTarget.type(), method));
110             // resource default is HttpMethod:protocol://url
111             if (methodMetadata == null) {
112                 result = methodHandler.invoke(args);
113             } else {
114                 String resourceName = methodMetadata.template().method().toUpperCase(Locale.getDefault())
115                     + ':'
116                     + hardCodedTarget.url()
117                     + methodMetadata.template().path();
118                 Entry entry = null;
119                 try {
120                     ContextUtil.enter(resourceName);
121                     entry = SphU.entry(resourceName, EntryType.OUT, 1, args);
122                     result = methodHandler.invoke(args);
123                 } catch (Throwable ex) {
124                     // fallback handle
125                     if (!BlockException.isBlockException(ex)) {
126                         Tracer.trace(ex);
127                     }
128                     if (fallbackFactory != null) {
129                         try {
130                             return fallbackMethodMap.get(method).invoke(fallbackFactory.create(ex), args);
131                         } catch (IllegalAccessException e) {
132                             // shouldn't happen as method is public due to being an
133                             // interface
134                             throw new AssertionError(e);
135                         } catch (InvocationTargetException e) {
136                             throw new AssertionError(e.getCause());
137                         }
138                     } else {
139                         // 若是R类型 并且不包含@FeignRetry 执行自动降级返回R
140                         FeignRetry feignRetry = AnnotationUtils.findAnnotation(method, FeignRetry.class);
141                         if (RestResponse.class == method.getReturnType() && Objects.isNull(feignRetry)) {
142                             log.error("feign 服务间调用异常", ex);
143                             return RestResponse.error(ex.getLocalizedMessage());
144                         } else {
145                             throw ex;
146                         }
147                     }
148                 } finally {
149                     if (entry != null) {
150                         entry.exit(1, args);
151                     }
152                     ContextUtil.exit();
153                 }
154             }
155         } else {
156             // other target type using default strategy
157             result = methodHandler.invoke(args);
158         }
159 
160         return result;
161     }
162 
163     @Override
164     public boolean equals(Object obj) {
165         if (obj instanceof com.alibaba.cloud.sentinel.feign.SentinelInvocationHandler) {
166             SentinelInvocationHandler other = (SentinelInvocationHandler) obj;
167             return target.equals(other.target);
168         }
169         return false;
170     }
171 
172     @Override
173     public int hashCode() {
174         return target.hashCode();
175     }
176 
177     @Override
178     public String toString() {
179         return target.toString();
180     }
181 }
